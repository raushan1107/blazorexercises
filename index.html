    public string Description { get; set; } = string.Empty;

TicketService Implementation
Business Logic
public class TicketService
{
    private readonly List<Ticket> tickets = new();
    private int nextId = 1;

    public event Action? OnChange;

    public List<Ticket> GetTickets() => tickets;

    public Ticket? GetById(int id) =>
        tickets.FirstOrDefault(t => t.Id == id);

    public void AddTicket(Ticket ticket)
    {
        ticket.Id = nextId++;
        ticket.CreatedAt = DateTime.Now;
        tickets.Add(ticket);
        NotifyStateChanged();
    }

    public void ResolveTicket(int id)
    {
        var ticket = GetById(id);
        if (ticket != null)
        {
            ticket.IsResolved = true;
            NotifyStateChanged();
        }
    }

    private void NotifyStateChanged() => OnChange?.Invoke();
}
ðŸ’¡ Singleton service shares state across all connected users in Blazor Server.
Tickets Page with Lifecycle
UI + Events
@page "/tickets"
@inject TicketService ticketService
@implements IDisposable

@foreach (var ticket in tickets)
{
    <TicketCard Ticket="ticket" />
}

@code {
    private List<Ticket> tickets = new();

    protected override void OnInitialized()
    {
        tickets = ticketService.GetTickets();
        ticketService.OnChange += Refresh;
    }

    private void Refresh()
    {
        InvokeAsync(() =>
        {
            tickets = ticketService.GetTickets();
            StateHasChanged();
        });
    }

    public void Dispose()
    {
        ticketService.OnChange -= Refresh;
    }
}
ðŸ’¡ InvokeAsync ensures UI updates occur on correct synchronization context.

Form Binding

    <InputText @bind-Value="ticket.Title" />

    <InputText @bind-Value="ticket.CreatedBy" />

</EditForm>

@code {
    private Ticket ticket = new();

    private void Save()
    {
        ticketService.AddTicket(ticket);
        nav.NavigateTo("/tickets");
    }
}
