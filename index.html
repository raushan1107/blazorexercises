    Lab 6 â€“ Real-Time Sync (Correct Implementation)

    Step 1: Create Services/TodoService.cs

    
public class TodoService
{
    public List Todos { get; } = new();
    public event Action? OnChange;

    public void Add(string task)
    {
        Todos.Add(task);
        NotifyStateChanged();
    }

    private void NotifyStateChanged()
    {
        OnChange?.Invoke();
    }
}

    Step 2: Register in Program.cs

    
builder.Services.AddSingleton();

    Step 3: Subscribe Safely in Todo.razor

    
@inject TodoService todoService
@implements IDisposable

protected override void OnInitialized()
{
    todoService.OnChange += HandleChange;
}

private async void HandleChange()
{
    await InvokeAsync(StateHasChanged);
}

public void Dispose()
{
    todoService.OnChange -= HandleChange;
}
